00368     public BaseAction subroutine(BaseAction orig) {
00369         // find root of orig
00370         BaseAction somewhereInOrig = orig;
00371 
00372         while (somewhereInOrig.getParent().getParent()
00373                 != null) {
00374             somewhereInOrig = somewhereInOrig.getParent();
00375         }
00376         somewhereInOrig.setParent(this);
00377         return orig;
00378     }
