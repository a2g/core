00285   private boolean update(double curTime) {
00286     /*
00287      * Save the run id. If the runId is incremented during this execution block,
00288      * we know that this run has been canceled.
00289      */
00290     final int curRunId = runId;
00291 
00292     boolean finished = curTime >= startTime + duration;
00293     if (isStarted && !finished) {
00294       // Animation is in progress.
00295       double progress = (curTime - startTime) / duration;
00296       onUpdate(interpolate(progress));
00297       return isRunning(curRunId); // Check if this run was canceled.
00298     }
00299     if (!isStarted && curTime >= startTime) {
00300       /*
00301        * Start the animation. We do not call onUpdate() because onStart() calls
00302        * onUpdate() by default.
00303        */
00304       isStarted = true;
00305       onStart();
00306       if (!isRunning(curRunId)) {
00307         // This run was canceled.
00308         return false;
00309       }
00310       // Intentional fall through to possibly end the animation.
00311     }
00312     if (finished) {
00313       // Animation is complete.
00314       isRunning = false;
00315       isStarted = false;
00316       onComplete();
00317       return false;
00318     }
00319     return true;
00320   }
