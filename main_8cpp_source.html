00034 QRect GetBoundingNonBlackRectangle(QImage image)
00035 {
00036     int max = image.width() > image.height()? image.width() : image.height();
00037     int top = 0;
00038     int left = 0;
00039     int bottom = image.height()-1;
00040     int right = image.width()-1;
00041     bool isToKeepAdvancingTop = true;
00042     bool isToKeepAdvancingLeft = true;
00043     bool isToKeepAdvancingBottom = true;
00044     bool isToKeepAdvancingRight = true;
00045     // this turns off the optimization :return QRect(QPoint(left,top),QPoint(right,bottom));
00046 
00047     // This is finding the bounding box by keeping shrinking the square (top,left,bottom,right)
00048     // until it runs into some non zero pixels.
00049     // we alternate between vertical and horizontal, because overall, this results in
00050     // fewer pixels to scan ( every horizontal line scanned can be two pixels shorter than the previous)
00051     for(int i=0;i<max;i++)
00052     {
00053         for(int k=left;k<=right && (isToKeepAdvancingTop || isToKeepAdvancingBottom);k++)
00054         {
00055             int colorTop = image.pixel(k,top)&RGB_MASK;
00056             int colorBottom = image.pixel(k,bottom)&RGB_MASK;
00057             isToKeepAdvancingTop &= (colorTop==0);
00058             isToKeepAdvancingBottom &= (colorBottom==0);
00059         }
00060 
00061         top += isToKeepAdvancingTop;
00062         bottom -= isToKeepAdvancingBottom;
00063 
00064         for(int k=top;k<=bottom && (isToKeepAdvancingLeft || isToKeepAdvancingRight);k++)
00065         {
00066             int colorLeft = image.pixel(left,k)&RGB_MASK;
00067             int colorRight = image.pixel(right,k)&RGB_MASK;
00068             isToKeepAdvancingRight &= (colorRight==0);
00069             isToKeepAdvancingLeft &= (colorLeft==0);
00070         }
00071 
00072         left += isToKeepAdvancingLeft;
00073         right -= isToKeepAdvancingRight;
00074     }
00075 
00076     QPoint topLeft(left,top);
00077     QPoint bottomRight(right, bottom);
00078 
00079     QRect toReturn(topLeft, bottomRight);
00080 
00081     return toReturn;
00082 }
