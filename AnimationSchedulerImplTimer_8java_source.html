00071   private static final int DEFAULT_FRAME_DELAY = 16;

00077   private static final int MIN_FRAME_DELAY = 5;

00082   private final List<AnimationHandleImpl> animationRequests = new ArrayList<AnimationHandleImpl>();

00087   private final Timer timer = new Timer() {
00088     @Override
00089     public void run() {

00094   @Override
00095   public AnimationHandle requestAnimationFrame(final AnimationCallback callback, Element element) {
00096     // Save the animation frame request.
00097     AnimationHandleImpl requestId = new AnimationHandleImpl(callback);
00098     animationRequests.add(requestId);
00099 
00100     // Start the timer if it isn't started.
00101     if (animationRequests.size() == 1) {
00102       timer.schedule(DEFAULT_FRAME_DELAY);
00103     }
00104 
00105     // Return the request id.
00106     return requestId;
00107   }

00114   private void cancelAnimationFrame(AnimationHandle requestId) {
00115     // Remove the request from the list.
00116     animationRequests.remove(requestId);
00117 
00118     // Stop the timer if there are no more requests.
00119     if (animationRequests.size() == 0) {
00120       timer.cancel();
00121     }
00122   }

00127   private void updateAnimations() {
00128     // Copy the animation requests to avoid concurrent modifications.
00129     AnimationHandleImpl[] curAnimations = new AnimationHandleImpl[animationRequests.size()];
00130     curAnimations = animationRequests.toArray(curAnimations);
00131 
00132     // Iterate over the animation requests.
00133     Duration duration = new Duration();
00134     for (AnimationHandleImpl requestId : curAnimations) {
00135       // Remove the current request.
00136       animationRequests.remove(requestId);
00137 
00138       // Execute the callback.
00139       requestId.getCallback().execute(duration.getStartMillis());
00140     }
00141 
00142     // Reschedule the timer if there are more animation requests.
00143     if (animationRequests.size() > 0) {
00144       /*
00145        * In order to achieve as close to 60fps as possible, we calculate the new
00146        * delay based on the execution time of this method. The delay will be
00147        * less than 16ms, assuming this method takes more than 1ms to complete.
00148        */
00149       timer.schedule(Math.max(MIN_FRAME_DELAY, DEFAULT_FRAME_DELAY - duration.elapsedMillis()));
00150     }
00151   }
